def sse(X, y, w):
    """Sum squared error function"""
    z = X.dot(w) - y
    return .5 * z.T.dot(z)

from functools import partial, update_wrapper

def wrapped_partial(func, *args, **kwargs):
    partial_func = partial(func, *args, **kwargs)
    update_wrapper(partial_func, func)
    return partial_func
